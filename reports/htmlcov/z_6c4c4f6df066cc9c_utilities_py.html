38def build_library_search_urls(author, title) -> dict[str, str]: 

39 """ 

40 Builds library search URLs using the given author and title. 

41 

42 This function takes the author's name and the title of a work, escapes 

43 them to make them URL-safe, and substitutes them into predefined search 

44 URL templates. The result is a dictionary of search URLs for external 

45 library systems or online catalog platforms. 

46 

47 :param author: The name of the author to search for (must be a string). 

48 :type author: str 

49 :param title: The title of the work to search for (must be a string). 

50 :type title: str 

51 :return: A dictionary where the keys are search platform identifiers 

52 and the values are URLs with the provided author and title 

53 embedded within them. 

54 :rtype: dict 

55 """ 

56 escaped_title = quote_plus(title, safe="") 

57 escaped_author = quote_plus(author, safe="") 

58 search_urls = { 

59 key: value.replace("{title}", escaped_title).replace("{author}", escaped_author) 

60 for key, value in _SEARCH_TEMPLATES.items() 

61 } 

62 return search_urls

65def render_icon(item, icon_mapping, span_id) -> Markup: 

66 """ 

67 Render an HTML span element with an icon based on the given item and mapping. 

68 

69 This function generates a span element containing a font-awesome icon 

70 corresponding to the provided item. The icon is determined by the 

71 ``icon_mapping`` dictionary. If the item is not found in the ``icon_mapping``, 

72 a default hidden placeholder HTML element is returned as a spacer. 

73 

74 :param item: The enumeration value representing the action (e.g., 'like', 'read') 

75 to map to a corresponding icon. 

76 :type item: str 

77 :param icon_mapping: A dictionary mapping enumeration values to font-awesome icon 

78 classes. 

79 :type icon_mapping: dict 

80 :param span_id: The ID attribute to assign to the span element in the resultant 

81 HTML string. 

82 :type span_id: str 

83 :return: A `Markup` object containing the rendered span element string 

84 with the appropriate icon or a placeholder if the item is not mapped. 

85 :rtype: Markup 

86 """ 

87 # item is the enumeration value (like 'like', 'read'), 

88 # icon_mapping is a dictionary mapping enum values to font-awesome icons 

89 if item in icon_mapping: 

90 return Markup( # nosec B704 

91 f'<span id="{span_id}">' + 

92 f'<i class="fa {icon_mapping[item]}" aria-hidden="true"></i>' + 

93 '</span>' 

94 ) 

95 # not in mapping, just use hidden default as a spacer 

96 return Markup(f'<span id="{span_id}">{PLACEHOLDER}</span>') # nosec B704

99def compute_next_url(request): 

100 """ 

101 Compute the next URL based on the referrer of the provided request object. 

102 

103 This function examines the `referrer` attribute of the given `request` object 

104 to compute the next URL to navigate to. If the `request` contains a valid `referrer`, 

105 the function extracts the path and appends the query string (if present). If no 

106 `referrer` is available, the function defaults to returning the URL for the 

107 "index" route. 

108 

109 :param request: The request object containing information about the current 

110 HTTP request. Should include a `referrer` attribute to extract navigation 

111 information from. 

112 :type request: flask.Request 

113 :return: The computed next URL string based on the `referrer`, or the default 

114 URL for the "index" route if no `referrer` is present. 

115 :rtype: str 

116 """ 

117 return urlparse(request.referrer).path + ( 

118 '?' + urlparse(request.referrer).query if urlparse(request.referrer).query else '') \ 

119 if request.referrer else url_for("index")

123def parse_url(url): 

124 """ 

125 Parses a given URL string and extracts its components. 

126 

127 This function takes a URL string as input, validates its presence, and 

128 parses it using Python's `urlparse` module. It extracts the path and query 

129 components and provides them along with the full URL in a dictionary format. 

130 

131 :param url: The URL string to be parsed. 

132 :type url: str 

133 :return: A dictionary containing the parsed components: "path", "query", 

134 and the original "full" URL string. Returns None if the input URL is 

135 not provided. 

136 :rtype: dict or None 

137 """ 

138 if not url: 

139 return None 

140 parsed = urlparse(url) 

141 return { 

142 "path": parsed.path, 

143 "query": parsed.query, 

144 "full": url 

145 }
