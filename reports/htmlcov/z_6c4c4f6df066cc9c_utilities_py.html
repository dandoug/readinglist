18def build_library_search_urls(author, title) -> dict[str, str]: 

19 """ 

20 Builds library search URLs using the given author and title. 

21 

22 This function takes the author's name and the title of a work, escapes 

23 them to make them URL-safe, and substitutes them into predefined search 

24 URL templates. The result is a dictionary of search URLs for external 

25 library systems or online catalog platforms. 

26 

27 :param author: The name of the author to search for (must be a string). 

28 :type author: str 

29 :param title: The title of the work to search for (must be a string). 

30 :type title: str 

31 :return: A dictionary where the keys are search platform identifiers 

32 and the values are URLs with the provided author and title 

33 embedded within them. 

34 :rtype: dict 

35 """ 

36 escaped_title = quote_plus(title, safe="") 

37 escaped_author = quote_plus(author, safe="") 

38 search_urls = { 

39 key: value.replace("{title}", escaped_title).replace("{author}", escaped_author) 

40 for key, value in _SEARCH_TEMPLATES.items() 

41 } 

42 return search_urls

45def render_icon(item, icon_mapping, span_id) -> Markup:
